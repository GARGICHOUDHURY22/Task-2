    <script>
        // Toggle visibility of the sections
        function toggleSection(sectionId) {
            const section = document.getElementById(sectionId);
            const isVisible = section.style.display === 'block';
            
            // Hide all sections first
            document.querySelectorAll('.subsection').forEach(sub => sub.style.display = 'none');
            
            // Toggle the clicked section visibility
            section.style.display = isVisible ? 'none' : 'block';
        }

        // Close sections if clicking outside
        document.addEventListener('click', function(event) {
            const mainSections = document.querySelector('.main-sections');
            const allSubsections = document.querySelectorAll('.subsection');
            
            // Check if the click was outside the main sections or subsections
            if (!mainSections.contains(event.target) && !Array.from(allSubsections).some(subsection => subsection.contains(event.target))) {
                // Hide all subsections
                allSubsections.forEach(sub => sub.style.display = 'none');
            }
        });
    </script>
